#!groovy
pipeline
{
	agent any
	stages
	{
		stage('clean')
		{
			steps
			{
				bat "mvn clean"
			}
		}
		stage('install')
		{
			steps
			{
				bat "mvn install" + " -Ptest -Denv=${params.apigee_env} -Dorg=${params.apigee_org} " +
						" -Dusername=${params.apigee_user} " +
						" -Dpassword=${params.apigee_password} " +
					" -Doptions=${params.apigee_options} "
						
			}
		}
	}
}
